  font-weight: bold;
  font-size: 25px;
`;

export const Counter: React.FC = () => {
  const [count, setCount] = useState<number>(0);
  return (
    <OuterDiv>
      <InnerDiv>
        <h1><i>-Counter-</i></h1>
        <H1>{count}</H1>
        <div>
          <Button onClick={() => setCount(count + 1)}>+</Button>
          <Button onClick={() => setCount(count - 1)}>-</Button>
        </div>
      </InnerDiv>
    </OuterDiv>
  );
};
